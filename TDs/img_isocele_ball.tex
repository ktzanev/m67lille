\documentclass[tikz,border=1pt]{standalone}
\usetikzlibrary{calc}
\usetikzlibrary{angles}
\usetikzlibrary{nicepoints}

\tikzset{
  eq/.style={sloped,transform shape,rotate=90}
}
\def\re[#1](#2)--(#3){(#3)--($(#3)!#1!(#2)$)--([turn]90:#1)--([turn]90:#1)}


\begin{document}
  \begin{tikzpicture}[rotate=35]
    \point["A"] at (1.8,3);
    \point["B"left] at (0,0);
    \point["C"right] at (3,0);
    \path
      ($(B)!.5!(C)$) [point="E"below]
      ($(E)!.5!90:(C)$) [point="D"{anchor=-21}]
      ($(B)!(D)!(A)$) [point="F"left]
      ($(C)!(D)!(A)$) [point="G"]
      ;
    \draw (A) -- (B) --node[eq]{$\sim$}  (E) --node[eq]{$\sim$} (C) -- cycle;
    \draw (D) -- (A) (D) -- node[eq ]{$=$} (B) (D) --node[eq]{$=$}  (C) (D) -- (E) (D) --node[eq]{$-$}  (F) (D) --node[eq]{$-$}  (G);
    \path
      pic[draw,double,angle radius = 0.42cm] {angle = F--A--D}
      pic[draw,double,angle radius = 0.35cm] {angle = D--A--G};
    \draw \re[.14cm](D)--(F) \re[.14cm](A)--(G) \re[.14cm](D)--(E);
  \end{tikzpicture}
\end{document}
