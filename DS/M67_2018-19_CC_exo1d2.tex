      (90:1)[point="A"'] coordinate(B) node[above=2pt,rectangle,scale=.84]{$A=B$}
      +(0:1)coordinate(X)
       (20:1)[point="A'"right]
       (280:1)[point="B'"below];

    \draw[fill=red, fill opacity=.2] (A) -- (M) -- (B') -- cycle;
    \draw[fill=blue,fill opacity=.2] (B) -- (M) -- (A') -- cycle;
    \draw[thick,red] (20:1) arc(20:90:1);
    \path
      pic[draw,fill=red,opacity=.7]{angle=A'--B'--A}
      pic[draw,fill=red,opacity=.7]{angle=A'--B--M}
      pic[draw,double]{angle=B--M--B'}
    ;
    \draw (M) -- (A) -- ([turn].5,0);
  \end{tikzpicture}
