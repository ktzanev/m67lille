\documentclass[tikz,border=7pt]{standalone}
\usetikzlibrary{nicepoints} % https://github.com/kpym/tikz-nicepoints
\usetikzlibrary{euclid.transforms} % https://github.com/kpym/tikz-euclid.transforms
\usetikzlibrary{angles,arrows.meta}
\begin{document}
  \begin{tikzpicture}
    \def\r{14mm}
    \draw[gray] circle(\r);
    \draw
      (180:\r) [point="A"left] --
      (60:\r) [point="B"] --
      (-60:\r) [point="C"below] -- cycle;
    \draw[thick, blue] (20:\r) [point="M"right] -- (A) (M) -- (B) (M) -- (C);
    \draw (B) -- ([rotation=C--B--A]M) [point="K"below];
    \fill[opacity=.14] (A)--(K)--(B) (C)--(M)--(B);
    \path[-{[scale=.7]Latex}] pic[angle radius=4mm,draw]{angle=A--B--C} pic[angle radius=3mm,draw]{angle=K--B--M};
    \path[angle radius=3mm] pic[draw]{angle=B--M--A} pic[draw]{angle=B--C--A};
    \path[angle radius=3.5mm] pic[draw,double]{angle=M--C--B} pic[draw,double]{angle=M--A--B};
  \end{tikzpicture}
\end{document}
